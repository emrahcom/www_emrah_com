# Parse querystring
function parsequery() {
    var query = {};
    var qarray = window.location.search.substring(1).split('&');
    for (var i in qarray) {
        var pair = qarray[i].split('=');
        query[decodeURIComponent(pair[0])] = decodeURIComponent(pair[1]);
    }

    return query
}


query = parsequery();
if (query['param'] != undefined) {
    alert(query['param']);
}
